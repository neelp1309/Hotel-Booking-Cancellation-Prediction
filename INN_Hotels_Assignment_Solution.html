# Data Preprocessing

## Missing Value Treatment
# Instead of removing or imputing missing values all at once, we will analyze each column and handle missing values iteratively.
# This approach is less memory-intensive and avoids potential crashes.

# Check for missing values
data_missing = INNHotelsGroup_df.isnull().sum()
missing_columns = data_missing[data_missing > 0].index.tolist()

# Handle missing values column-wise
for col in missing_columns:
    if INNHotelsGroup_df[col].dtype == 'object':
        # Impute missing categorical values with the mode
        mode_value = INNHotelsGroup_df[col].mode()[0]
        INNHotelsGroup_df[col].fillna(mode_value, inplace=True)
    else:
        # Impute missing numerical values with the median (robust to outliers)
        median_value = INNHotelsGroup_df[col].median()
        INNHotelsGroup_df[col].fillna(median_value, inplace=True)

## Outlier Detection and Treatment
### Updated Approach:
# Use a selective treatment approach, processing only critical numeric columns iteratively to save memory.

numeric_columns = INNHotelsGroup_df.select_dtypes(include=['float64', 'int64']).columns

for col in numeric_columns:
    if INNHotelsGroup_df[col].nunique() > 10:  # Process only continuous variables
        Q1 = INNHotelsGroup_df[col].quantile(0.25)
        Q3 = INNHotelsGroup_df[col].quantile(0.75)
        IQR = Q3 - Q1
        lower_bound = Q1 - 1.5 * IQR
        upper_bound = Q3 + 1.5 * IQR

        # Clip values outside bounds
        INNHotelsGroup_df[col] = INNHotelsGroup_df[col].clip(lower=lower_bound, upper=upper_bound)

## Feature Engineering
### Updated Approach:
# Create new features incrementally and validate memory usage after each step.

INNHotelsGroup_df['total_nights'] = INNHotelsGroup_df['no_of_week_nights'] + INNHotelsGroup_df['no_of_weekend_nights']
INNHotelsGroup_df['price_per_person'] = INNHotelsGroup_df['avg_price_per_room'] / (INNHotelsGroup_df['no_of_adults'] + INNHotelsGroup_df['no_of_children'].replace(0, 1))
INNHotelsGroup_df['booking_lead_time'] = INNHotelsGroup_df['lead_time'] / INNHotelsGroup_df['total_nights'].replace(0, 1)

## Data Scaling
### Updated Approach:
# Apply scaling selectively on columns used in model building, reducing memory overhead.
from sklearn.preprocessing import MinMaxScaler

scaler = MinMaxScaler()
columns_to_scale = ['lead_time', 'avg_price_per_room', 'total_nights', 'price_per_person', 'booking_lead_time']

encoded_data_test = ohe.transform(X_test[categorical_columns])
encoded_df_test = pd.DataFrame(encoded_data_test, columns=encoded_feature_names, index=X_test.index)
